Chapter 2. C 프로그래밍의 기초

3. 연산자 (Operator)

데이터를 가공하고 계산하기 위한 연산자의 종류를 배우고, 우선순위와 결합 법칙을 통해 복합적인 수식을 이해합니다.

Think & Objective

"컴퓨터는 연산의 순서를 어떻게 결정할까요?"

  • 연산, 비교, 논리 등의 여러 종류의 연산자의 의미를 알고 프로그래밍에 적절하게 사용할 수 있다.
  • 다양한 연산자를 활용하여 프로그래밍 할 수 있다.

산술 연산자란?

가장 기본이 되는 연산자로, 수학의 사칙연산과 유사합니다. 나머지 연산(%)은 프로그래밍에서 홀짝 구분이나 배수 판별에 자주 쓰이니 꼭 기억하세요!

문법 포인트: 나누기와 나머지

  • / (나누기): 정수끼리 나누면 결과도 정수입니다. (예: 5 / 2 = 2)
  • % (나머지): 나눈 후 남은 값을 구합니다. (예: 5 % 2 = 1)

우선순위 주의

곱셈(*), 나눗셈(/), 나머지(%) 연산이 덧셈(+), 뺄셈(-)보다 먼저 계산됩니다. 우선순위를 바꾸려면 괄호( )를 사용하세요.

Step 1

기초 사칙연산

#include <stdio.h>

int main() {
    int a = 10, b = 3;

    printf("덧셈: %d + %d = %d\n", a, b, a + b);    // 13
    printf("뺄셈: %d - %d = %d\n", a, b, a - b);    // 7
    printf("곱셈: %d * %d = %d\n", a, b, a * b);    // 30
    printf("나눗셈: %d / %d = %d\n", a, b, a / b);  // 3 (소수점 버림)
    printf("나머지: %d %% %d = %d\n", a, b, a % b); // 1 (나눈 나머지)

    return 0;
}

대입 및 증감 연산자

단순히 값을 넣는 = 외에도 기존 값에 연산을 더하는 복합 대입 연산자가 있습니다. 또한 ++나 --를 사용하면 값을 1씩 아주 간편하게 늘리거나 줄일 수 있습니다.

복합 대입 연산자 예시
a += 1
(a = a + 1 과 같음)
a *= 2
(a = a * 2 와 같음)
전위 vs 후위 (어려움 주의!)
  • ++a (전위): 1을 먼저 더하고 문장을 실행합니다.
  • a++ (후위): 문장을 실행한 후 나중에 1을 더합니다.
Step 2

증감 및 복합 연산 실습

#include <stdio.h>

int main() {
    int score = 10;

    score += 5; // score = 10 + 5
    printf("복합 대입 결과: %d\n", score); // 15

    score++;    // score = 15 + 1
    printf("1 증가 결과: %d\n", score); // 16

    score--;    // score = 16 - 1
    printf("1 감소 결과: %d\n", score); // 15

    return 0;
}

비교 및 논리 연산자

조건문에서 "두 값이 같은가?" 혹은 "A이고 B인가?"와 같은 질문을 던질 때 사용합니다. C언어에서 참은 1, 거짓은 0으로 표현된다는 점이 매우 중요합니다.

비교 연산자 (결과는 0 또는 1)

== 양쪽 값이 같다
!= 양쪽 값이 다르다
>, <= 크다, 작거나 같다
Step 3

참/거짓 출력하기

#include <stdio.h>

int main() {
    int a = 10, b = 20;

    printf("a == b 결과: %d\n", a == b); // 0
    printf("a < b 결과: %d\n", a < b);   // 1
    printf("a != b 결과: %d\n", a != b); // 1

    return 0;
}

논리 연산자 (조건 결합)

&& (AND) 둘 다 참일 때만 참
|| (OR) 둘 중 하나라도 참이면 참
! (NOT) 참/거짓을 뒤집음
Step 4

복합 조건문 이해

#include <stdio.h>

int main() {
    int a = 10;
    // a가 0보다 크고(&&) 20보다 작은가?
    printf("0 < a < 20: %d\n", (a > 0) && (a < 20)); // 1
    return 0;
}

증감 연산자 (++, --)

예제 전치 vs 후치 연산의 차이
88.c 
#include <stdio.h>
int main()
{
	int i = 13, j = 5, temp,
		a = 10, b = 20, c = 30, d = 40;
	temp = ++i; printf("%d %d\n", temp, i);
	temp = i++; printf("%d %d\n", temp, i);
	temp = --j; printf("%d %d\n", temp, j);
	temp = j--; printf("%d %d\n", temp, j);
	++a; b++; --c; d--;
	printf("%d %d %d %d", a, b, c, d);
	return 0;
}

예제 논리 연산자
90-1.c 
#include <stdio.h>
int main()
{
	int i = 13, j = 5, temp;
	printf("%d %d\n", i == 13 && j == 5, i == 13 && j == 4);

예제 단락 회로 평가(Short-Circuit)
90-2.c 
#include <stdio.h>
int main()
{
	int i = 13, j = 5;
	if (i < 10 && ++j > 3) { } // i < 10이 거짓이므로 ++j는 실행되지 않음!
	printf("%d %d\n", i, j);   // 결과: 13 5
	return 0;
}
왜 j가 6이 아니라 5인가요?
A && B (AND 연산)

앞의 조건(A)이 거짓(0)이면, 뒤의 조건(B)은 결과와 상관없이 **아예 실행하지 않습니다.**

A || B (OR 연산)

앞의 조건(A)이 참(1)이면, 뒤의 조건(B)은 결과와 상관없이 **아예 실행하지 않습니다.**

i < 10 (F) && ++j > 3 결과: 거짓(0)

"이미 앞쪽에서 거짓으로 결론이 났기 때문에, 뒤쪽의 ++j는 아예 쳐다보지도 않습니다."

비트 연산자

예제 비트 논리 연산자 (&, |, ^, ~)

#include <stdio.h>
int main()
{
	int i = 13;  // 00000000 00000000 00000000 00001101

	printf("%d %d %d %d", i&j, i | j, i^j, ~i);

}

{

}

연산자 응용 예제 종합

예제 정기예금 이자 계산기
92.c 
#include <stdio.h>
int main()
{
	int money, duration;
	double rate;
	double result;
	printf("정기예금 금액을 입력해 주세요 :");
	scanf("%d", &money);
	printf("정기예금 이율을 입력해 주세요 :");
	scanf("%lf", &rate);
	printf("정기예금 기간을 입력해 주세요 :");
	scanf("%d", &duration);
	result = money + money * duration * 0.01 * rate;
	printf("%d원을 %.1f%%이율로 %d년간 예치 후 원리 합계는 %.0f입니다.",
		money, rate, duration, result);
	return 0;
}
예제 사칙연산 계산
93.c 
#include <stdio.h>
int main()
{
	int a, b;
	printf("두 수를 입력하세요 :");
	scanf("%d %d", &a, &b);
	printf("%d + %d = %d\n", a, b, a + b);
	printf("%d - %d = %d\n", a, b, a - b);
	printf("%d * %d = %d\n", a, b, a * b);
	printf("%d / %d = %f\n", a, b, (float)(a / b));
	printf("%d %% %d = %d" , a, b, a % b);
	return 0;
}
