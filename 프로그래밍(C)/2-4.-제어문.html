Chapter 2. C 프로그래밍의 기초

4. 제어문 (Control Statement)

프로그램의 흐름을 내 마음대로! 조건에 따라 다른 길을 가거나, 똑같은 작업을 반복하는 제어 구조를 단계적으로 마스터합니다.

학습 목표 (Objective)

  • 순차, 선택, 반복의 제어 구조를 이해하고 예제로 직접 활용할 수 있다.
  • 중첩 제어 구조(제어문 안의 제어문)를 다양한 패턴에 적용할 수 있다.

순차 구조

위에서 아래로 정해진 순서에 따라 차례대로 코드가 실행되는 가장 기본적인 흐름 구조입니다.

선택 구조

주어진 조건의 참·거짓에 따라 서로 다른 코드를 실행하는 분기 구조입니다.

반복 구조

특정 조건이 만족될 때까지 동일한 코드를 계속 반복 실행하여 작업을 자동화하는 구조입니다.

조건문 (Selection)

프로그램은 위에서 아래로 순차적으로 실행되지만, 조건문을 사용하면 상황에 따라 실행할 코드를 선택할 수 있습니다. 특정 조건이 참(1)일 때만 코드가 실행되도록 흐름을 제어합니다.

if-else 문법 상세

  • 1. if(조건식): 소괄호 안의 조건이 참이면 중괄호 { } 안의 내용을 실행합니다.
  • 2. else if: 앞의 조건이 거짓일 때, 다른 조건을 추가로 확인합니다.
  • 3. else: 제시한 모든 조건이 거짓일 때 실행되는 '나머지' 구역입니다.
Step 1

기초 조건문 (양수/음수 판별)

#include <stdio.h>

int main() {
    int num = 10;

    if (num > 0) {
        printf("이 숫자는 양수입니다.\n");
    } else if (num < 0) {
        printf("이 숫자는 음수입니다.\n");
    } else {
        printf("이 숫자는 0입니다.\n");
    }

    return 0;
}
Step 2

switch-case 문 (메뉴 선택)

* switch문은 여러 가지 경우(case) 중 하나를 선택할 때 if문보다 가독성이 좋습니다.

#include <stdio.h>

int main() {
    int choice = 2;

    switch (choice) {
        case 1:
            printf("콜라를 선택했습니다.\n");
            break; // break를 쓰지 않으면 다음 case까지 실행됩니다!
        case 2:
            printf("사이다를 선택했습니다.\n");
            break;
        default: // 위에 해당하지 않는 모든 경우
            printf("잘못된 선택입니다.\n");
    }
    return 0;
}

선택 구조 (단순 if 문)

단순 if 문은 조건식이 참일 때만 특정 코드를 실행하는 구조입니다. 조건이 거짓이면 if 블록 안의 코드는 생략되고 다음 코드로 넘어갑니다.

학생들이 가장 많이 하는 실수 BEST 1

수학에서의 ‘같다’는 = 이지만, C 언어의 조건문에서 ‘같다’는 반드시 ==를 사용해야 합니다. =는 대입 연산자입니다.

오류: if (score = 100) { ... } // score에 100을 대입하고, 조건식 결과가 항상 참이 되어버림
정답: if (score == 100) { ... } // score가 100인지 비교
예제 단순 if문 (참/거짓 판별)
98.c 
#include <stdio.h>
int main()
{
    int i = 13, j = 5;
    if (-1) putchar('A');
    if (1) putchar('B');
    if (13) putchar('C');
    if (0) putchar('D');
    if (i > j)
    {
        putchar('E');
        putchar('F');
    }
    if (i < j)
        putchar('G');
    putchar('H');
    return 0;
}

선택 구조 (if-else 문)

if-else 문은 조건이 참일 때와 거짓일 때 실행되는 문장이 서로 다를 때 사용하는 선택문입니다.

예제if ~ else문 (합격/불합격)
99.c 
#include <stdio.h>
int main()
{
    int standard = 80, score = 76;
    if (standard < score)
    {
        putchar('P');
    }
    else
    {
        putchar('F');
    }
    return 0;
}

선택 구조 (다중 if 문)

다중 if 문은 if 문 안에 또 다른 if 문을 포함하는 형태로, 비교나 판단 조건을 복합적으로 사용할 수 있습니다.

예제 다중 if문 (평균 비교)
100.c 
#include <stdio.h>
int main()
{
    int standard = 173, height = 168;
    if (standard > height)
    {
        printf("평균 이하");
    }
    else if (standard == height)
    {
        printf("평균");
    }
    else
    {
        printf("평균 이상");
    }
    return 0;
}

선택 구조 (switch-case 문)

비교해야 할 조건이 많을 때 다중 if 문 대신 switch-case 문을 사용하면 하나의 조건으로 여러 case 중 하나를 선택해 실행할 수 있어 코드를 더 직관적으로 만들 수 있습니다.

예제 switch문 (학점 계산)
101.c 
#include <stdio.h>
int main()
{
    int score;
    char grade;
    puts("점수를 입력하세요(1~100)");
    putchar('>');
    scanf("%d", &score);
    switch (score / 10)
    {
      case 10:
      case 9: grade = 'A'; break;
      case 8: grade = 'B'; break;
      case 7: grade = 'C'; break;
      case 6: grade = 'D'; break;
      default: grade = 'F';
    }
    printf("학점은 %c 입니다.", grade);
    return 0;
}

반복문 (Iteration)

동일한 작업을 수천, 수만 번 직접 타이핑할 필요가 없습니다. 반복문은 정해진 횟수나 특정 조건이 만족될 때까지 코드를 반복해서 실행해 줍니다.

for 문

반복 횟수가 명확할 때 (10번 반복 등)

횟수 중심
for문 횟수가 명확할 때
for_basic.c 
int sum = 0;
// for(초기식; 조건식; 증감식)
for (int i = 1; i <= 10; i++) {
    sum = sum + i;
}
printf("1~10 합: %d", sum);

while 문

특정 조건이 유지되는 동안 (게임이 끝날 때까지 등)

조건 중심
while문 조건이 중요할 때
while_basic.c 
int i = 1, sum = 0;
// 조건식이 참인 동안 반복
while (i <= 10) {
    sum += i;
    i++; // 증감식 필수!
}
printf("1~10 합: %d", sum);

무한 루프(Infinite Loop) 주의!

while 문에서 반복 조건을 변화시키는 코드를 빼먹으면 조건이 영원히 참이 되어 프로그램이 멈추지 않을 수 있습니다. 

int i = 0;
while (i < 5) {
    printf("안녕하세요\n");
    // i++; (이 코드가 없으면 i는 계속 0이라 무한 반복됨)
}

반복 구조 (for 문)

for 문은 가장 자주 사용하는 반복문으로, 반복 횟수를 정확히 알고 있을 때 사용하면 편리합니다.

예제 for문 (구구단 2단)
103.c 
#include <stdio.h>
int main()
{
    int i;
    for (i = 1; i < 10; i++)
    {
        printf("2 x %d = %2d\n", i, i * 2);
    }
    return 0;
}

반복 구조 (while 문)

while 문은 정확한 반복 횟수를 알 수 없고, 주어진 조건이 참인지 계속 확인하면서 반복해야 할 때 사용합니다.

예제 while문 (0 입력시 종료)
104-1.c 
#include <stdio.h>
int main()
{
    int answer = 1;
    while (answer != 0)
    {
        scanf("%d", &answer);
        printf("%d 입력\n", answer);
    }
    return 0;
}
예제 while문 (입력한 수까지 출력)
104-2.c 
#include <stdio.h>
int main()
{
    int i = 1, end;
    printf("수를 입력하세요 : ");
    scanf("%d", &end);
    while (i < end + 1) {
        printf("%d ", i);
        i++;
    }
    return 0;
}

반복 구조 (do-while 문)

do-while 문은 기본적으로 while 문과 비슷하지만, 조건의 참·거짓과 상관없이 먼저 한 번은 실행한 뒤에 조건을 검사한다는 점이 다릅니다.

예제 do ~ while문 (조건 확인 나중)
105-1.c 
#include <stdio.h>
int main()
{
    int answer;
    do {
        scanf("%d", &answer);
        printf("%d 입력\n", answer);
    } while (answer != 0);
    return 0;
}
예제 do ~ while문 (소문자 -> 대문자 변환)
105-2.c 
#include <stdio.h>
int main()
{
    char ch;
    do {
        scanf(" %c", &ch);
        printf("입력한 소문자 %c는 대문자 %c입니다.\n", ch, ch - 32);
    } while (ch != 'q');
    return 0;
}

반복 흐름 제어문 (Iteration Control Statement)

반복문을 실행하다가 강제로 빠져나오거나, 특정 부분만 건너뛰고 싶을 때 break와 continue를 사용합니다.

break (즉시 탈출)

조건을 만족하면 반복문을 그 즉시 완전히 종료합니다. 

for (int i = 1; i <= 10; i++) {
    if (i == 5) break;
    printf("%d ", i);
}
// 결과: 1 2 3 4

continue (건너뛰기)

이어지는 명령문을 무시하고 다음 회차로 넘어갑니다. 

for (int i = 1; i <= 5; i++) {
    if (i == 3) continue;
    printf("%d ", i);
}
// 결과: 1 2 4 5

반복 흐름 제어문 (break 문)

break 문과 continue 문은 조건문이나 반복문의 실행 흐름을 제어하는 명령문입니다.

예제 break 문
106.c 
#include <stdio.h>
int main()
{
    int answer;
    while (1)
    {
        scanf("%d", &answer);
        if (answer == 0)
        {
            break;
        }
        else {
            printf("%d 입력", answer);
        }
    }
    return 0;
}

반복 흐름 제어문 (continue 문)

continue 문은 현재 반복을 건너뛰고 다음 반복으로 넘어가도록 흐름을 제어하는 명령문입니다.

예제 for문과 continue문 (107-1.c)
107-1.c 
#include <stdio.h>
int main()
{
    int i, count = 0, answer;
    scanf("%d", &answer);
    for (i = 1; i < 101; i++)
    {
        if (i % answer != 0)
        {
            continue;
        }
        count++;
    }
    printf("1부터 100까지 수 중에서 %d의 배수는 %d개입니다.",
        answer, count);
    return 0;
}
예제 while문과 continue문 (107-2.c)
107-2.c 
#include <stdio.h>
int main()
{
    int i = 0;
    while (i++ < 10) {
        if (i == 7) continue;
        printf("%d ", i);
    }
    return 0;
}

중첩 제어 구조 (Nested Control Structure)

제어문 안에 또 다른 제어문이 포함된 구조입니다. 바깥쪽 루프가 한 번 실행될 때 안쪽 루프는 지정된 횟수만큼 모두 실행됩니다.

중첩 for문 (구구단)

108.c

2단부터 3단까지 구구단을 출력합니다. 바깥쪽 for문은 단(i)을, 안쪽 for문은 곱해지는 수(j)를 의미합니다. 

#include <stdio.h>
int main()
{
    int i, j;
    for (i = 2; i < 4; i++)
    {
        for (j = 1; j < 10; j++)
        {
            printf("%d x %d = %2d\n", i, j, i * j);
        }
        puts("");
    }
    return 0;
}

숫자 직각삼각형 출력

111.c

중첩 반복문을 사용하여 줄이 바뀔 때마다 출력되는 숫자의 개수를 늘려가며 직각삼각형 모양을 만듭니다. 

#include <stdio.h>
int main()
{
    int i, j, count = 1;
    for (i = 0; i < 5; i++)
    {
        for (j = 0; j < i; j++)
        {
            printf("%d ", count++);
        }
        printf("\n");
    }
    return 0;
}

실전 예제 (중첩 제어 구조)

PPT 수록 예제

제어문 안에 또 다른 제어문이 들어가는 것을 중첩 제어 구조라고 합니다. 교과서에 수록된 핵심 예제를 직접 실행해 봅시다.

예제 1. 소수 판별 프로그램

1과 자기 자신만으로 나누어 떨어지는 수인지 판별합니다. (break문 활용)

prime_number.c 
#include <stdio.h>

int main() {
    int num, i;
    printf("소수인지 판별할 정수를 입력하세요: ");
    scanf("%d", &num);

    // 2부터 입력한 숫자 바로 앞(num-1)까지 나누어 봅니다.
    for (i = 2; i < num; i++) {
        if (num % i == 0) {
            break; // 중간에 나누어 떨어지면 소수가 아님! 즉시 반복문 탈출
        }
    }

    if (i == num) {
        printf("%d는 소수입니다.\n", num);
    } else {
        printf("%d는 소수가 아닙니다.\n", num);
    }
    return 0;
}
예제 2. 완전수(Perfect Number) 구하기

자신을 제외한 약수의 합이 자신과 같은 수 (예: 6 = 1 + 2 + 3). 1부터 10000 사이의 완전수를 찾습니다.

perfect_number.c 
#include <stdio.h>

int main() {
    int i, j, sum;

    printf("1 ~ 10000 사이의 완전수:\n");

    // 1부터 10000까지 반복 (바깥쪽 루프)
    for (i = 1; i <= 10000; i++) {
        sum = 0; // 매 숫자마다 약수의 합을 0으로 초기화

        // 약수를 구하기 위한 반복 (안쪽 루프 - 중첩 제어 구조)
        for (j = 1; j < i; j++) {
            if (i % j == 0) {
                sum += j; // 나누어 떨어지면 약수이므로 합 누적
            }
        }

        // 자신을 제외한 약수의 합이 자신과 같다면
        if (i == sum) {
            printf("%d ", i);
        }
    }
    printf("\n");
    return 0;
}
예제 3. 중첩 for문 (도형 출력)

이중 for 문을 활용하여 도형을 출력해봅시다.

102.c 
#include <stdio.h>
int main()
{
    int i, j, k;
    for (i = 0; i < 10; i++)
    {

        }
        puts("");

}

제어문 심화 코드

두 수 사이의 누적합 구하기 (109-1.c)

109-1.c

입력받은 start부터 end까지의 모든 정수를 더하여 누적합을 계산하는 프로그램입니다. for문의 초기식 부분을 생략하여 작성한 것이 특징입니다. 

#include<stdio.h>
int main()
{
    int start, end, tempStart, sum = 0;
    printf("두 수를 입력하세요 : ");
    scanf("%d %d", &start, &end);
    tempStart = start;
    for (; start < end + 1; start++)
    {
        sum += start;
    }
    printf("%d부터 %d까지의 합은 %d입니다.", tempStart, end, sum);
    return 0;
}

소수(Prime Number) 판별 (109-2.c)

109-2.c

완전수(Perfect Number) 판단 프로그램 (110.c)

110.c

완전수란?

자신을 제외한 약수의 합으로 나타낼 수 있는 수입니다.
예) 6의 약수는 1, 2, 3, 6이고 자신(6)을 제외한 약수의 합 (1+2+3)이 6과 같으므로 6은 완전수입니다.

1부터 10000 사이의 수 중에서 중첩 for문을 활용하여 모든 완전수를 찾아 출력하는 심화 프로그램입니다. 

#include <stdio.h>
int main()
{
    int i, j, sum=0, count=0;
    for (i=1; i<10000; i++) {
        for (j = 1; j < i; j++) {
              if (i%j == 0) {
                sum += j;
              }
         }
        if(i == sum) printf("%d번째 완전수: %d\n", ++count, i);
        sum = 0;
      }
      return 0;
}
